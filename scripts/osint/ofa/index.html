#!/usr/bin/env bash
# ═══════════════════════════════════════════════════════════════════
# osint_fraud_analyzer.sh · SecFerro Division Series · v1.0.0
# Advanced IP/Domain fraud intelligence aggregator
# Sources: ipinfo.io · ip-api.com · AbuseIPDB · Shodan
#          VirusTotal · StopForumSpam
# Author: Anonymousik — anonymousik.is-a.dev/scripts/osint
# Legal: Evidence gathering for law enforcement reports ONLY
# ═══════════════════════════════════════════════════════════════════
## (Termux: set -euo pipefail disabled for compatibility)
## (Standard Bash: uncomment the line below)
#set -euo pipefail

# ── CONFIG (via environment variables — never hardcode keys) ────────
ABUSEIPDB_KEY="${ABUSEIPDB_KEY:-}"
VIRUSTOTAL_KEY="${VIRUSTOTAL_KEY:-}"
SHODAN_KEY="${SHODAN_KEY:-}"
IPINFO_TOKEN="${IPINFO_TOKEN:-}"
OUTPUT_DIR="${OUTPUT_DIR:-./fraud_reports}"
REPORT_FILE=""

# ── COLORS ──────────────────────────────────────────────────────────
RED='\033[0;31m'; YELLOW='\033[1;33m'; GREEN='\033[0;32m'
CYAN='\033[0;36m'; BOLD='\033[1m'; NC='\033[0m'

# ── DEPENDENCY CHECK ────────────────────────────────────────────────
for cmd in curl jq date; do
    command -v "$cmd" &>/dev/null || { echo "[!] Missing dep: $cmd" >&2; exit 1; }
done

# ── INPUT VALIDATION (sanitize: only IP/domain-safe chars) ──────────
TARGET="${1:-}"
[[ -z "$TARGET" ]] && { echo "Użycie: $0 <ip|domena>" >&2; exit 1; }
if ! [[ "$TARGET" =~ ^[a-zA-Z0-9._:-]+$ ]]; then
    echo "[!] Invalid target format: '${TARGET}'" >&2; exit 1
fi

# ── HELPER: safe_curl (validates JSON, checks HTTP 200) ─────────────
safe_curl() {
    local url="$1" response http_code body
    response=$(curl -sS -w "\n%{http_code}" --max-time 15 \
        -H "User-Agent: FraudAnalyzer/2.0" "$url" 2>/dev/null) || return 1
    http_code=$(echo "$response" | tail -n1)
    body=$(echo "$response" | head -n -1)
    [[ "$http_code" == "200" ]] || return 1
    echo "$body" | jq empty 2>/dev/null || return 1
    echo "$body"
}

log_section() { echo -e "\n${CYAN}${BOLD}══════════════════════════════${NC}";
                echo -e "${CYAN}${BOLD}  $1${NC}";
                echo -e "${CYAN}${BOLD}══════════════════════════════${NC}"; }
append_report() { echo "$1" >> "$REPORT_FILE"; }

# ── SETUP OUTPUT DIR ────────────────────────────────────────────────
mkdir -p "$OUTPUT_DIR"
TIMESTAMP=$(date +"%Y%m%d_%H%M%S")
REPORT_FILE="${OUTPUT_DIR}/fraud_report_${TARGET//\//_}_${TIMESTAMP}.txt"
{
    echo "======================================"
    echo "  FRAUD INTELLIGENCE REPORT"
    echo "  Target : $TARGET"
    echo "  Date   : $(date -u '+%Y-%m-%d %H:%M:%S UTC')"
    echo "======================================"
} > "$REPORT_FILE"

echo -e "${BOLD}[*] Target: ${YELLOW}${TARGET}${NC}"

# ════════════════════════════════════════════════════════════════════
# MODULE 1: GEO + ASN — ipinfo.io
# ════════════════════════════════════════════════════════════════════
log_section "1/6 GEO + ASN — ipinfo.io"
append_report $'\n[MODULE 1] GEO + ASN'

API_URL="https://ipinfo.io/${TARGET}/json"
[[ -n "$IPINFO_TOKEN" ]] && API_URL+="?token=${IPINFO_TOKEN}"
if GEO=$(safe_curl "$API_URL"); then
    echo "$GEO" | jq '{IP:.ip,Hostname:(.hostname//"N/A"),
        Org:(.org//"N/A"),City:(.city//"N/A"),Country:.country,Bogon:(.bogon//false)}'
    append_report "$(echo "$GEO" | jq .)"
else
    echo -e "${YELLOW}[!] ipinfo.io unavailable or rate limit${NC}"
fi

# ════════════════════════════════════════════════════════════════════
# MODULE 2: VPN / TOR / PROXY — ip-api.com
# ════════════════════════════════════════════════════════════════════
log_section "2/6 VPN/TOR/PROXY — ip-api.com"
append_report $'\n[MODULE 2] VPN/TOR/PROXY'

FIELDS="status,country,regionName,city,isp,org,as,proxy,hosting,query"
if IPAPI=$(safe_curl "http://ip-api.com/json/${TARGET}?fields=${FIELDS}"); then
    IS_PROXY=$(echo "$IPAPI" | jq -r '.proxy')
    IS_HOST=$(echo "$IPAPI" | jq -r '.hosting')
    echo "$IPAPI" | jq '{ISP:.isp,Org:.org,AS:.as,Is_Proxy:.proxy,Is_DC:.hosting}'
    [[ "$IS_PROXY" == "true" ]] && echo -e "${RED}[!!!] PROXY/VPN DETECTED!${NC}"
    [[ "$IS_HOST" == "true" ]] && echo -e "${YELLOW}[!] IP belongs to DATACENTER${NC}"
    append_report "$(echo "$IPAPI" | jq .)"
fi

# ════════════════════════════════════════════════════════════════════
# MODULE 3: ABUSE HISTORY — AbuseIPDB
# ════════════════════════════════════════════════════════════════════
log_section "3/6 ABUSE HISTORY — AbuseIPDB"
if [[ -z "$ABUSEIPDB_KEY" ]]; then
    echo -e "${YELLOW}[!] ABUSEIPDB_KEY not set — skipping${NC}"
else
    if ABUSE=$(curl -sS --max-time 15 \
        -H "Key: ${ABUSEIPDB_KEY}" -H "Accept: application/json" \
        "https://api.abuseipdb.com/api/v2/check?ipAddress=${TARGET}&maxAgeInDays=90&verbose"); then
        SCORE=$(echo "$ABUSE" | jq -r '.data.abuseConfidenceScore // 0')
        REPORTS=$(echo "$ABUSE" | jq -r '.data.totalReports // 0')
        echo "$ABUSE" | jq '.data | {Score:.abuseConfidenceScore,Reports:.totalReports}'
        (( SCORE >= 80 )) && echo -e "${RED}[!!!] HIGH RISK: ${SCORE}/100 (${REPORTS} reports)${NC}"
        (( SCORE >= 30 )) && (( SCORE < 80 )) && echo -e "${YELLOW}[!] MEDIUM: ${SCORE}/100${NC}"
        (( SCORE < 30 )) && echo -e "${GREEN}[+] LOW risk: ${SCORE}/100${NC}"
    fi
fi

# ════════════════════════════════════════════════════════════════════
# MODULE 4: OPEN PORTS / CVE — Shodan
# ════════════════════════════════════════════════════════════════════
log_section "4/6 OPEN PORTS / CVE — Shodan"
if [[ -n "$SHODAN_KEY" ]]; then
    if SHODAN=$(safe_curl \
        "https://api.shodan.io/shodan/host/${TARGET}?key=${SHODAN_KEY}"); then
        echo "$SHODAN" | jq '{OS:(.os//"unknown"),Ports:.ports,Tags:.tags,CVEs:(.vulns|keys?//[])}'
        append_report "$(echo "$SHODAN" | jq .)"
    else
        echo -e "${YELLOW}[!] Shodan: no data for this IP${NC}"
    fi
fi

# ════════════════════════════════════════════════════════════════════
# MODULE 5: MALWARE / PHISHING — VirusTotal
# ════════════════════════════════════════════════════════════════════
log_section "5/6 MALWARE/PHISHING — VirusTotal"
if [[ -n "$VIRUSTOTAL_KEY" ]]; then
    if VT=$(curl -sS --max-time 15 \
        -H "x-apikey: ${VIRUSTOTAL_KEY}" \
        "https://www.virustotal.com/api/v3/ip_addresses/${TARGET}"); then
        echo "$VT" | jq '.data.attributes | {Malicious:.last_analysis_stats.malicious,
            Suspicious:.last_analysis_stats.suspicious,Harmless:.last_analysis_stats.harmless}'
        MAL=$(echo "$VT" | jq -r '.data.attributes.last_analysis_stats.malicious // 0')
        (( MAL > 0 )) && echo -e "${RED}[!!!] ${MAL} AV engines: MALICIOUS${NC}"
    fi
fi

# ════════════════════════════════════════════════════════════════════
# MODULE 6: SPAM REPUTATION — StopForumSpam
# ════════════════════════════════════════════════════════════════════
log_section "6/6 SPAM REPUTATION — StopForumSpam"
if SFS=$(safe_curl "https://api.stopforumspam.org/api?ip=${TARGET}&json"); then
    echo "$SFS" | jq '{Listed:.ip.appears,Frequency:.ip.frequency,Confidence:.ip.confidence}'
    LISTED=$(echo "$SFS" | jq -r '.ip.appears // 0')
    (( LISTED > 0 )) && echo -e "${RED}[!!!] IP listed in SPAM/FRAUD database!${NC}"
    append_report "$(echo "$SFS" | jq .)"
fi

# ════════════════════════════════════════════════════════════════════
# SUMMARY + LEGAL REPORTING CHANNELS
# ════════════════════════════════════════════════════════════════════
log_section "SUMMARY & LEGAL CHANNELS"
{
    echo "======================================"
    echo "  LEGAL REPORTING CHANNELS"
    echo "======================================"
    echo "PL - CERT Polska:     https://incydent.cert.pl"
    echo "PL - Policja:         https://www.policja.pl"
    echo "EU - Europol EC3:     https://europol.europa.eu/report-a-crime"
    echo "Community AbuseIPDB:  https://www.abuseipdb.com/report"
    echo "Google Safe Browsing: https://safebrowsing.google.com"
    echo "Evidence: $REPORT_FILE"
} | tee -a "$REPORT_FILE"

echo -e "\n${GREEN}${BOLD}[+] Analysis complete. Report: ${REPORT_FILE}${NC}"
